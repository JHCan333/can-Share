```

<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>捕获右键粘贴事件</title>
</head>

<input id="test"/>
<script>
    var isRight = false

    // 通过监听鼠标右击事件还有粘贴事件，来判断是否是 右键粘贴事件
    document.getElementById('test').addEventListener('mousedown',rightEvent)
    document.getElementById('test').addEventListener('paste',pasetEvent)

    //判断是否是右键点击
    function rightEvent (event) {
        event = window.event || event
        if (event.button == 2) {
            isRight = true
        }
    }

    //执行粘贴事件时执行的方法
    function pasetEvent (event) {
        let eve = window.event || event
        let target = eve.target || eve.srcElement
        if (isRight == true) {

        }
    }
</script>
```
